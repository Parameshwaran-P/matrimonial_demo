🗄️ Database Schema (PostgreSQL)

-- USERS (authentication)
users {
  id UUID PRIMARY KEY,
  phone VARCHAR UNIQUE NOT NULL,
  email VARCHAR UNIQUE,
  password_hash VARCHAR,
  role ENUM('user','admin'),
  is_verified BOOLEAN DEFAULT false,
  otp_secret VARCHAR,
  created_at TIMESTAMP,
  last_active TIMESTAMP
}

-- PROFILES (matrimonial details)
profiles {
  id UUID → users.id,
  full_name VARCHAR,
  date_of_birth DATE,
  gender ENUM('male','female','other'),
  height_cm INT,
  weight_kg INT,
  religion VARCHAR, caste VARCHAR, sub_caste VARCHAR,
  mother_tongue VARCHAR,
  city VARCHAR, state VARCHAR, country VARCHAR,
  education VARCHAR, college VARCHAR,
  profession VARCHAR, employer VARCHAR,
  annual_income_range VARCHAR,
  marital_status ENUM,
  about_me TEXT,
  family_type ENUM, family_values ENUM,
  horoscope_details JSONB,
  status ENUM('pending','approved','rejected','deactivated'),
  admin_notes TEXT,
  profile_score INT,
  photo_urls TEXT[],
  is_photo_blurred BOOLEAN DEFAULT false,
  created_at TIMESTAMP, updated_at TIMESTAMP
}

-- INTERESTS
interests {
  id UUID, from_user UUID, to_user UUID,
  status ENUM('pending','accepted','rejected','withdrawn'),
  message TEXT,
  sent_at TIMESTAMP, responded_at TIMESTAMP,
  UNIQUE(from_user, to_user)
}

-- MESSAGES (only between accepted interests)
messages {
  id UUID, conversation_id UUID,
  sender_id UUID, content TEXT (encrypted),
  is_read BOOLEAN, is_deleted BOOLEAN,
  created_at TIMESTAMP
}

-- REPORTS & BLOCKS
reports { id, reporter_id, reported_id, reason, status, created_at }
blocks  { id, blocker_id, blocked_id, created_at }

-- PRIVACY SETTINGS
privacy_settings {
  user_id UUID, hide_phone BOOLEAN,
  hide_email BOOLEAN, hide_lastname BOOLEAN,
  blur_photo BOOLEAN, show_online BOOLEAN,
  appear_in_search BOOLEAN, profile_visible BOOLEAN
}

🔐 Security Architecture

  • OTP Authentication — Phone-based OTP via MSG91/Twilio, 5-min expiry, rate-limited
  • JWT Tokens — Short-lived access tokens (15min) + refresh tokens (30 days), stored httpOnly cookies
  • RBAC — Role-based access: user, admin, super_admin with middleware enforcement
  • Message Encryption — AES-256 at-rest, TLS in-transit for all messages
  • Contact Privacy — Phone/email only revealed post mutual interest via server-side logic
  • Media Security — Photos stored in private S3 bucket, served via signed URLs (1h expiry)
  • Rate Limiting — Express-rate-limit on auth endpoints, 5 OTP requests/hour/IP
  • Input Validation — Zod schemas for all API inputs, XSS/SQL injection prevention
  • GDPR Compliance — Right to erasure, data export, consent management
  • Audit Logging — All admin actions logged with actor, timestamp, and action details

🧮 Compatibility Scoring Algorithm

// Weighted scoring across 6 dimensions
function calculateCompatibility(userA, userB) {
  const weights = {
    community:   0.25,  // Religion, caste, community
    lifestyle:   0.20,  // Values, diet, smoking/drinking
    education:   0.15,  // Qualification level match
    location:    0.15,  // City, state proximity
    family:      0.15,  // Family type, values
    horoscope:   0.10   // Guna match (optional)
  };
  
  const scores = {
    community:  matchCommunity(userA, userB),
    lifestyle:  matchLifestyle(userA, userB),
    education:  matchEducation(userA, userB),
    location:   matchLocation(userA, userB),
    family:     matchFamily(userA, userB),
    horoscope:  matchHoroscope(userA, userB)
  };
  
  return Object.entries(weights)
    .reduce((total, [key, w]) => total + (scores[key] * w), 0);
}
